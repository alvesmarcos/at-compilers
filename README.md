## Atividade 01 (26/01/2017)
Implementação de um trecho de código nos três níveis de abstração.

**C++**
```cpp
int main(){
	int x = 3;
	int y = 1;
	int z = 0;

	if(x==y)
		z = 0;
	z = x + y * 5 - 2;
	return 0;
}
```

**Assembly GAS/AT&T**

 ```assembly

.SECTION .data
	z:
		.int 0
	y:
		.int 1

.SECTION .bss
	.lcomm	x, 4

.SECTION .text
	.globl _start

_start:
	movl 	$3, x
	movl 	(x), %ecx

	cmpl 	(y), %ecx
	jne 	.L01
	movl 	$0, (z)

.L01:
	movl 	(y), %eax

	imull 	$5, %eax

	addl 	(x), %eax
	subl 	$2, %eax

	movl 	%eax, (z)

_exit:
	movl 	$1, %eax
	movl 	(z), %ebx
	int 	$0x80

 ```
